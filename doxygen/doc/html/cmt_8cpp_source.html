62 void track(cv::Mat im_prev, cv::Mat im_gray, const std::vector<std::pair<cv::KeyPoint, int> >& keypointsIN, std::vector<std::pair<cv::KeyPoint, int> >& keypointsTracked, std::vector<unsigned char>& status, int THR_FB)
63 {
64  //Status of tracked keypoint - True means successfully tracked
65  status = std::vector<unsigned char>();
66  //for(int i = 0; i < keypointsIN.size(); i++)
67  // status.push_back(false);
68  //If at least one keypoint is active
69  if(keypointsIN.size() > 0)
70  {
71  std::vector<cv::Point2f> pts;
72  std::vector<cv::Point2f> pts_back;
73  std::vector<cv::Point2f> nextPts;
74  std::vector<unsigned char> status_back;
75  std::vector<float> err;
76  std::vector<float> err_back;
77  std::vector<float> fb_err;
78  for(size_t i = 0; i < keypointsIN.size(); i++)
79  pts.push_back(cv::Point2f(keypointsIN[i].first.pt.x,keypointsIN[i].first.pt.y));
80 
81  //Calculate forward optical flow for prev_location
82  cv::calcOpticalFlowPyrLK(im_prev, im_gray, pts, nextPts, status, err);
83  //Calculate backward optical flow for prev_location
84  cv::calcOpticalFlowPyrLK(im_gray, im_prev, nextPts, pts_back, status_back, err_back);
85 
86  //Calculate forward-backward error
87  for(size_t i = 0; i < pts.size(); i++)
88  {
89  cv::Point2f v = pts_back[i]-pts[i];
90  fb_err.push_back(sqrt(v.dot(v)));
91  }
92 
93  //Set status depending on fb_err and lk error
94  for(size_t i = 0; i < status.size(); i++)
95  status[i] = (fb_err[i] <= THR_FB) & status[i];
96 
97  keypointsTracked = std::vector<std::pair<cv::KeyPoint, int> >();
98  for(size_t i = 0; i < pts.size(); i++)
99  {
100  std::pair<cv::KeyPoint, int> p = keypointsIN[i];
101  if(status[i])
102  p.first.pt = nextPts[i];
103  keypointsTracked.push_back(p);
104  }
105  }
106  else keypointsTracked = std::vector<std::pair<cv::KeyPoint, int> >();
107 }

132 void CMT::initialise(cv::Mat im_gray0, cv::Point2f topleft, cv::Point2f bottomright)
133 {
134 
135  //Initialise detector, descriptor, matcher
136  detector = cv::Algorithm::create<cv::FeatureDetector>(detectorType.c_str());
137  descriptorExtractor = cv::Algorithm::create<cv::DescriptorExtractor>(descriptorType.c_str());
138  descriptorMatcher = cv::DescriptorMatcher::create(matcherType.c_str());
139  std::vector<std::string> list;
140  cv::Algorithm::getList(list);
141 
142  //Get initial keypoints in whole image
143  std::vector<cv::KeyPoint> keypoints;
144  detector->detect(im_gray0, keypoints);
145 
146  //Remember keypoints that are in the rectangle as selected keypoints
147  std::vector<cv::KeyPoint> selected_keypoints;
148  std::vector<cv::KeyPoint> background_keypoints;
149  inout_rect(keypoints, topleft, bottomright, selected_keypoints, background_keypoints);
150  descriptorExtractor->compute(im_gray0, selected_keypoints, selectedFeatures);
151 
152  if(selected_keypoints.size() == 0)
153  {
154  printf("No keypoints found in selection");
155  return;
156  }
157 
158  //Remember keypoints that are not in the rectangle as background keypoints
159  cv::Mat background_features;
160  descriptorExtractor->compute(im_gray0, background_keypoints, background_features);
161 
162  //Assign each keypoint a class starting from 1, background is 0
163  selectedClasses = std::vector<int>();
164  for(size_t i = 1; i <= selected_keypoints.size(); i++)
165  selectedClasses.push_back(i);
166  std::vector<int> backgroundClasses;
167  for(size_t i = 0; i < background_keypoints.size(); i++)
168  backgroundClasses.push_back(0);
169 
170  //Stack background features and selected features into database
171  featuresDatabase = cv::Mat(background_features.rows+selectedFeatures.rows, std::max(background_features.cols,selectedFeatures.cols), background_features.type());
172  if(background_features.cols > 0)
173  background_features.copyTo(featuresDatabase(cv::Rect(0,0,background_features.cols, background_features.rows)));
174  if(selectedFeatures.cols > 0)
175  selectedFeatures.copyTo(featuresDatabase(cv::Rect(0,background_features.rows,selectedFeatures.cols, selectedFeatures.rows)));
176 
177  //Same for classes
178  classesDatabase = std::vector<int>();
179  for(size_t i = 0; i < backgroundClasses.size(); i++)
180  classesDatabase.push_back(backgroundClasses[i]);
181  for(size_t i = 0; i < selectedClasses.size(); i++)
182  classesDatabase.push_back(selectedClasses[i]);
183 
184  //Get all distances between selected keypoints in squareform and get all angles between selected keypoints
185  squareForm = std::vector<std::vector<float> >();
186  angles = std::vector<std::vector<float> >();
187  for(size_t i = 0; i < selected_keypoints.size(); i++)
188  {
189  std::vector<float> lineSquare;
190  std::vector<float> lineAngle;
191  for(size_t j = 0; j < selected_keypoints.size(); j++)
192  {
193  float dx = selected_keypoints[j].pt.x-selected_keypoints[i].pt.x;
194  float dy = selected_keypoints[j].pt.y-selected_keypoints[i].pt.y;
195  lineSquare.push_back(sqrt(dx*dx+dy*dy));
196  lineAngle.push_back(atan2(dy, dx));
197  }
198  squareForm.push_back(lineSquare);
199  angles.push_back(lineAngle);
200  }
201 
202  //Find the center of selected keypoints
203  cv::Point2f center(0,0);
204  for(size_t i = 0; i < selected_keypoints.size(); i++)
205  center += selected_keypoints[i].pt;
206  center *= (1.0/selected_keypoints.size());
207 
208  //Remember the rectangle coordinates relative to the center
209  centerToTopLeft = topleft - center;
210  centerToTopRight = cv::Point2f(bottomright.x, topleft.y) - center;
211  centerToBottomRight = bottomright - center;
212  centerToBottomLeft = cv::Point2f(topleft.x, bottomright.y) - center;
213 
214  //Calculate springs of each keypoint
215  springs = std::vector<cv::Point2f>();
216  for(size_t i = 0; i < selected_keypoints.size(); i++)
217  springs.push_back(selected_keypoints[i].pt - center);
218 
219  //Set start image for tracking
220  im_prev = im_gray0.clone();
221 
222  //Make keypoints 'active' keypoints
223  activeKeypoints = std::vector<std::pair<cv::KeyPoint,int> >();
224  for(size_t i = 0; i < selected_keypoints.size(); i++)
225  activeKeypoints.push_back(std::make_pair(selected_keypoints[i], selectedClasses[i]));
226 
227  //Remember number of initial keypoints
228  nbInitialKeypoints = selected_keypoints.size();
229 }

269 float findMinSymetric(const std::vector<std::vector<float> >& dist, const std::vector<bool>& used, int limit, int &i, int &j)
270 {
271  float min = dist[0][0];
272  i = 0;
273  j = 0;
274  for(int x = 0; x < limit; x++)
275  {
276  if(!used[x])
277  {
278  for(int y = x+1; y < limit; y++)
279  if(!used[y] && dist[x][y] <= min)
280  {
281  min = dist[x][y];
282  i = x;
283  j = y;
284  }
285  }
286  }
287  return min;
288 }

290 std::vector<Cluster> linkage(const std::vector<cv::Point2f>& list)
291 {
292  float inf = 10000000;0;
293  std::vector<bool> used;
294  for(size_t i = 0; i < 2*list.size(); i++)
295  used.push_back(false);
296  std::vector<std::vector<float> > dist;
297  for(size_t i = 0; i < list.size(); i++)
298  {
299  std::vector<float> line;
300  for(size_t j = 0; j < list.size(); j++)
301  {
302  if(i == j)
303  line.push_back(inf);
304  else
305  {
306  cv::Point2f p = list[i]-list[j];
307  line.push_back(sqrt(p.dot(p)));
308  }
309  }
310  for(size_t j = 0; j < list.size(); j++)
311  line.push_back(inf);
312  dist.push_back(line);
313  }
314  for(size_t i = 0; i < list.size(); i++)
315  {
316  std::vector<float> line;
317  for(size_t j = 0; j < 2*list.size(); j++)
318  line.push_back(inf);
319  dist.push_back(line);
320  }
321  std::vector<Cluster> clusters;
322  while(clusters.size() < list.size()-1)
323  {
324  int x, y;
325  float min = findMinSymetric(dist, used, list.size()+clusters.size(), x, y);
326  Cluster cluster;
327  cluster.first = x;
328  cluster.second = y;
329  cluster.dist = min;
330  cluster.num = (x < (int)list.size() ? 1 : clusters[x-list.size()].num) + (y < (int)list.size() ? 1 : clusters[y-list.size()].num);
331  used[x] = true;
332  used[y] = true;
333  int limit = list.size()+clusters.size();
334  for(int i = 0; i < limit; i++)
335  {
336  if(!used[i])
337  dist[i][limit] = dist[limit][i] = std::min(dist[i][x], dist[i][y]);
338  }
339  clusters.push_back(cluster);
340  }
341  return clusters;
342 }

344 void fcluster_rec(std::vector<int>& data, const std::vector<Cluster>& clusters, float threshold, const Cluster& currentCluster, int& binId)
345 {
346  int startBin = binId;
347  if(currentCluster.first >= (int)data.size())
348  fcluster_rec(data, clusters, threshold, clusters[currentCluster.first - data.size()], binId);

360  binId++;

398 void CMT::estimate(const std::vector<std::pair<cv::KeyPoint, int> >& keypointsIN, cv::Point2f& center, float& scaleEstimate, float& medRot, std::vector<std::pair<cv::KeyPoint, int> >& keypoints)
399 {
400  center = cv::Point2f(NAN,NAN);
401  scaleEstimate = NAN;
402  medRot = NAN;
403 
404  //At least 2 keypoints are needed for scale
405  if(keypointsIN.size() > 1)
406  {
407  //sort
408  std::vector<PairInt> list;
409  for(size_t i = 0; i < keypointsIN.size(); i++)
410  list.push_back(std::make_pair(keypointsIN[i].second, i));
411  std::sort(&list[0], &list[0]+list.size(), comparatorPair<int>);
412  for(size_t i = 0; i < list.size(); i++)
413  keypoints.push_back(keypointsIN[list[i].second]);
414 
415  std::vector<int> ind1;
416  std::vector<int> ind2;
417  for(size_t i = 0; i < list.size(); i++)
418  for(size_t j = 0; j < list.size(); j++)
419  {
420  if(i != j && keypoints[i].second != keypoints[j].second)
421  {
422  ind1.push_back(i);
423  ind2.push_back(j);
424  }
425  }
426  if(ind1.size() > 0)
427  {
428  std::vector<int> class_ind1;
429  std::vector<int> class_ind2;
430  std::vector<cv::KeyPoint> pts_ind1;
431  std::vector<cv::KeyPoint> pts_ind2;
432  for(size_t i = 0; i < ind1.size(); i++)
433  {
434  class_ind1.push_back(keypoints[ind1[i]].second-1);
435  class_ind2.push_back(keypoints[ind2[i]].second-1);
436  pts_ind1.push_back(keypoints[ind1[i]].first);
437  pts_ind2.push_back(keypoints[ind2[i]].first);
438  }
439 
440  std::vector<float> scaleChange;
441  std::vector<float> angleDiffs;
442  for(size_t i = 0; i < pts_ind1.size(); i++)
443  {
444  cv::Point2f p = pts_ind2[i].pt - pts_ind1[i].pt;
445  //This distance might be 0 for some combinations,
446  //as it can happen that there is more than one keypoint at a single location
447  float dist = sqrt(p.dot(p));
448  float origDist = squareForm[class_ind1[i]][class_ind2[i]];
449  scaleChange.push_back(dist/origDist);
450  //Compute angle
451  float angle = atan2(p.y, p.x);
452  float origAngle = angles[class_ind1[i]][class_ind2[i]];
453  float angleDiff = angle - origAngle;
454  //Fix long way angles
455  if(fabs(angleDiff) > CV_PI)
456  angleDiff -= sign(angleDiff) * 2.0F * (float)CV_PI;
457  angleDiffs.push_back(angleDiff);
458  }
459  scaleEstimate = median(scaleChange);
460  if(!estimateScale)
461  scaleEstimate = 1;
462  medRot = median(angleDiffs);
463  if(!estimateRotation)
464  medRot = 0;
465  votes = std::vector<cv::Point2f>();
466  for(size_t i = 0; i < keypoints.size(); i++)
467  votes.push_back(keypoints[i].first.pt - scaleEstimate * rotate(springs[keypoints[i].second-1], medRot));
468  //Compute linkage between pairwise distances
469  std::vector<Cluster> linkageData = linkage(votes);
470 
471  //Perform hierarchical distance-based clustering
472  std::vector<int> T = fcluster(linkageData, (float)thrOutlier);
473  //Count votes for each cluster
474  std::vector<int> cnt = binCount(T);
475  //Get largest class
476  int Cmax = argmax(cnt);
477 
478  //Remember outliers
479  outliers = std::vector<std::pair<cv::KeyPoint, int> >();
480  std::vector<std::pair<cv::KeyPoint, int> > newKeypoints;
481  std::vector<cv::Point2f> newVotes;
482  for(size_t i = 0; i < keypoints.size(); i++)
483  {
484  if(T[i] != Cmax)
485  outliers.push_back(keypoints[i]);
486  else
487  {
488  newKeypoints.push_back(keypoints[i]);
489  newVotes.push_back(votes[i]);
490  }
491  }
492  keypoints = newKeypoints;
493 
494  center = cv::Point2f(0,0);
495  for(size_t i = 0; i < newVotes.size(); i++)
496  center += newVotes[i];
497  center *= (1.0/newVotes.size());
498  }
499  }
500 }

520 bool CMT::processFrame(cv::Mat im_gray)
521 {
522  trackedKeypoints = std::vector<std::pair<cv::KeyPoint, int> >();
523  std::vector<unsigned char> status;
524  track(im_prev, im_gray, activeKeypoints, trackedKeypoints, status);
525 
526  cv::Point2f center;
527  float scaleEstimate;
528  float rotationEstimate;
529  std::vector<std::pair<cv::KeyPoint, int> > trackedKeypoints2;
530  estimate(trackedKeypoints, center, scaleEstimate, rotationEstimate, trackedKeypoints2);
531  trackedKeypoints = trackedKeypoints2;
532 
533  //Detect keypoints, compute descriptors
534  std::vector<cv::KeyPoint> keypoints;
535  cv::Mat features;
536  detector->detect(im_gray, keypoints);
537  descriptorExtractor->compute(im_gray, keypoints, features);
538 
539  //Create list of active keypoints
540  activeKeypoints = std::vector<std::pair<cv::KeyPoint, int> >();
541 
542  //For each keypoint and its descriptor
543  for(size_t i = 0; i < keypoints.size(); i++)
544  {
545  cv::KeyPoint keypoint = keypoints[i];
546 
547  //First: Match over whole image
548  //Compute distances to all descriptors
549  std::vector<cv::DMatch> matches;
550  descriptorMatcher->match(featuresDatabase,features.row(i), matches);
551 
552  //Convert distances to confidences, do not weight
553  std::vector<float> combined;
554  for(size_t j = 0; j < matches.size(); j++)
555  combined.push_back(1 - matches[j].distance / descriptorLength);
556 
557  std::vector<int>& classes = classesDatabase;
558 
559  //Sort in descending order
560  std::vector<PairFloat> sorted_conf;
561  for(size_t j = 0; j < combined.size(); j++)
562  sorted_conf.push_back(std::make_pair(combined[j], j));
563  std::sort(&sorted_conf[0], &sorted_conf[0]+sorted_conf.size(), comparatorPairDesc<float>);
564 
565  //Get best and second best index
566  int bestInd = sorted_conf[0].second;
567  int secondBestInd = sorted_conf[1].second;
568 
569  //Compute distance ratio according to Lowe
570  float ratio = (1-combined[bestInd]) / (1-combined[secondBestInd]);
571 
572  //Extract class of best match
573  int keypoint_class = classes[bestInd];
574 
575  //If distance ratio is ok and absolute distance is ok and keypoint class is not background
576  if(ratio < thrRatio && combined[bestInd] > thrConf && keypoint_class != 0)
577  activeKeypoints.push_back(std::make_pair(keypoint, keypoint_class));
578 
579  //In a second step, try to match difficult keypoints
580  //If structural constraints are applicable
581  if(!(isnan(center.x) | isnan(center.y)))
582  {
583  //Compute distances to initial descriptors
584  std::vector<cv::DMatch> matches;
585  descriptorMatcher->match(selectedFeatures, features.row(i), matches);
586 
587  //Convert distances to confidences
588  std::vector<float> confidences;
589  for(size_t j = 0; j < matches.size(); j++)
590  confidences.push_back(1 - matches[j].distance / descriptorLength);
591 
592  //Compute the keypoint location relative to the object center
593  cv::Point2f relative_location = keypoint.pt - center;
594 
595  //Compute the distances to all springs
596  std::vector<float> displacements;
