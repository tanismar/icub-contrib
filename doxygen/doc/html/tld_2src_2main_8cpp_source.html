179  bool respond(const Bottle &command, Bottle &reply)
180  {
181  int on=Vocab::encode("on");
182  int off=Vocab::encode("off");
183  int ack=Vocab::encode("ack");
184  int nack=Vocab::encode("nack");
185 
186  if (command.size()>0)
187  {
188  switch (command.get(0).asVocab())
189  {
190  //-----------------
191  case VOCAB4('l','o','a','d'):
192  {
193  if (command.size()>=2)
194  {
195  if (command.get(1).isString())
196  {
197  mutex.lock();
198  string model=rf->findFileByName(command.get(1).asString()).c_str();
199  tracker->readFromFile(model.c_str());
200  mutex.unlock();
201 
202  reply.addVocab(ack);
203  reply.addString(model.c_str());
204  }
205  else if (command.size()>=5)
206  {
207  mutex.lock();
208  boundingBox.x=command.get(1).asInt();
209  boundingBox.y=command.get(2).asInt();
210  boundingBox.width=command.get(3).asInt()-boundingBox.x;
211  boundingBox.height=command.get(4).asInt()-boundingBox.y;
212  loadBoundingBox=true;
213  mutex.unlock();
214 
215  reply.addVocab(ack);
216  }
217  else
218  reply.addVocab(nack);
219  }
220  else
221  reply.addVocab(nack);
222 
223  return true;
224  }
225 
226  //-----------------
227  case VOCAB4('s','a','v','e'):
228  {
229  if (command.size()>=2)
230  {
231  mutex.lock();
232  string model=rf->getHomeContextPath().c_str();
233  model+="/";
234  model+=command.get(1).asString().c_str();
235  tracker->writeToFile(model.c_str());
236  mutex.unlock();
237 
238  reply.addVocab(ack);
239  reply.addString(model.c_str());
240  }
241  else
242  reply.addVocab(nack);
243 
244  return true;
245  }
246 
247  //-----------------
248  case VOCAB4('s','t','a','r'):
249  {
250  mutex.lock();
251  doTLD=true;
252  mutex.unlock();
253 
254  reply.addVocab(ack);
255  return true;
256  }
257 
258  //-----------------
259  case VOCAB4('s','t','o','p'):
260  {
261  mutex.lock();
262  doTLD=false;
263  mutex.unlock();
264 
265  reply.addVocab(ack);
266  return true;
267  }
268 
269  //-----------------
270  case VOCAB4('c','l','e','a'):
271  {
272  mutex.lock();
273  tracker->release();
274  mutex.unlock();
275 
276  reply.addVocab(ack);
277  return true;
278  }
279 
280  //-----------------
281  case VOCAB3('s','e','t'):
282  {
283  if (command.size()>=3)
284  {
285  int subcmd=command.get(1).asVocab();
286  int mode=command.get(2).asVocab();
287  switch (subcmd)
288  {
289  //-----------------
290  case VOCAB4('l','e','a','r'):
291  {
292  if ((mode==on) || (mode==off))
293  {
294  mutex.lock();
295  tracker->learningEnabled=(mode==on);
296  mutex.unlock();
297  reply.addVocab(ack);
298  }
299  else
300  reply.addVocab(nack);
301 
302  break;
303  }
304 
305  //-----------------
306  case VOCAB4('a','l','t','e'):
307  {
308  if ((mode==on) || (mode==off))
309  {
310  mutex.lock();
311  tracker->alternating=(mode==on);
312  mutex.unlock();
313  reply.addVocab(ack);
314  }
315  else
316  reply.addVocab(nack);
317 
318  break;
319  }
320 
321  //-----------------
322  default:
323  reply.addVocab(nack);
324  }
325  }
326  else
327  reply.addVocab(nack);
328 
329  return true;
330  }
331 
332  //-----------------
333  case VOCAB3('g','e','t'):
334  {
335  if (command.size()>=2)
336  {
337  int subcmd=command.get(1).asVocab();
338  switch (subcmd)
339  {
340  //-----------------
341  case VOCAB4('l','e','a','r'):
342  {
343  reply.addVocab(ack);
344  reply.addVocab(tracker->learningEnabled?on:off);
345  break;
346  }
347 
348  //-----------------
349  case VOCAB4('a','l','t','e'):
350  {
351  reply.addVocab(ack);
352  reply.addVocab(tracker->alternating?on:off);
353  break;
354  }
355 
356  //-----------------
357  default:
358  reply.addVocab(nack);
359  }
360  }
361  else
362  reply.addVocab(nack);
363 
364  return true;
365  }
366 
367  //-----------------
368  default:
369  return RFModule::respond(command,reply);
370  }
371  }
372 
373  reply.addVocab(nack);
374  return true;
375  }

384  bool updateModule()
385  {
386  ImageOf<PixelBgr> *img=imgInPort.read();
387  if (img==NULL)
388  return true;
389 
390  mutex.lock();
391  cv::Mat imgMat=cv::cvarrToMat((IplImage*)img->getIplImage());
392 
393  if (initDetectorCascade)
394  {
395  ImageOf<PixelMono> imgMono;
396  imgMono.resize(img->width(),img->height());
397 
398  cv::Mat imgMonoMat=cv::cvarrToMat((IplImage*)imgMono.getIplImage());
399  cv::cvtColor(imgMat,imgMonoMat,CV_BGR2GRAY);
400 
401  tracker->detectorCascade->imgWidth=imgMonoMat.cols;
402  tracker->detectorCascade->imgHeight=imgMonoMat.rows;
403  tracker->detectorCascade->imgWidthStep=imgMonoMat.step;
404  initDetectorCascade=false;
405  }
406 
407  bool skipProcessingOnce=false;
408  if (loadBoundingBox)
409  {
410  ImageOf<PixelMono> imgMono;
411  imgMono.resize(img->width(),img->height());
412 
413  cv::Mat imgMonoMat=cv::cvarrToMat((IplImage*)imgMono.getIplImage());
414  cv::cvtColor(imgMat,imgMonoMat,CV_BGR2GRAY);
415 
416  tracker->selectObject(imgMonoMat,&boundingBox);
417  skipProcessingOnce=true;
418  loadBoundingBox=false;
419  }
420 
421  if (doTLD && !skipProcessingOnce)
422  {
423  tracker->processImage(imgMat); // yep, color image, otherwise it does not work
424  if (tracker->currBB!=NULL)
425  {
426  cv::Point tl,br;
427  tl.x=tracker->currBB->x;
428  tl.y=tracker->currBB->y;
429  br.x=tl.x+tracker->currBB->width;
430  br.y=tl.y+tracker->currBB->height;
431 
432  if (dataOutPort.getOutputCount()>0)
433  {
434  Bottle &data=dataOutPort.prepare();
435  data.clear();
436 
437  data.addInt(tl.x);
438  data.addInt(tl.y);
439  data.addInt(br.x);
440  data.addInt(br.y);
441  data.addDouble(tracker->currConf);
442 
443  dataOutPort.write();
444  }
445 
446  if (imgOutPort.getOutputCount()>0)
447  cv::rectangle(imgMat,tl,br,(tracker->currConf>=0.5)?cv::Scalar(255,0,0):cv::Scalar(0,255,255),2);
448  }
449  }
450 
451  if (imgOutPort.getOutputCount()>0)
452  {
453  imgOutPort.prepare()=*img;
454  imgOutPort.write();
455  }
456 
457  mutex.unlock();
458  return true;
459  }
